static int GenerateCombinationsHelper(int[] S, int length, int index, 
int[] currentComb, int startIndex, List <int[]> combinations)
{
    int flag = 0;
    flag += 1;
    if (index == length)
    {
            int[] comb = new int[length];
            Array.Copy(currentComb, comb, length);
            combinations.Add(comb);
            return flag;
        }
        flag += 2;
        for (int i = startIndex; i < S.Length; i++)
        {
            currentComb[index] = S[i];
            flag += 5;
            flag += GenerateCombinationsHelper(S, length, index + 1, currentComb, i + 1, combinations);
        }
        return flag;
    }
}
